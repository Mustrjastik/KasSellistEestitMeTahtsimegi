add_namespace = bombing

country_event = {
    id = bombing.1
    title = bombing_event.t
    desc = bombing.1.d
    fire_only_once = yes
    trigger = {
    AND = {
	        has_idea = nato_airstrikes_USA
	OR = {
	USA = {
	country_exists = no
	}
	any_owned_state = {
	anti_air_building > 1
	}	}
	}
	}
    option = {
        name = bombing.1.a
        remove_ideas = nato_airstrikes_USA
    }
}
